prob = pulp.LpProblem("UTA_Method_with_Binary_Inconsistency_Solver", pulp.LpMinimize)

def estimation_constraints(number_of_alternatives, problem):

    # each alternative has over and under estimation error variable
    over = pulp.LpVariable.dicts("over", range(number_of_alternatives))
    under = pulp.LpVariable.dicts("under", range(number_of_alternatives))

    # minimize the sum of over and under estimation error
    problem += pulp.lpSum([over[i] + under[i] for i in range(number_of_alternatives)])

    for i in range(number_of_alternatives):
        problem += over[i] >= 0
        problem += under[i] >= 0
    return problem

def criterion_values_constraints(alternatives, vars, problem, gains):
    # nonnegativity constraints
    for var in vars.values():
        problem += var >= 0

    # get maximum and minimum values of each criterion
    min_values = get_min_values(alternatives)
    max_values = get_max_values(alternatives)
    alpha = dict()
    beta = dict()
    for criterion in alternatives.columns:
        if gains[criterion]:
            alpha[criterion] = min_values[criterion]
            beta[criterion] = max_values[criterion]
        else:
            alpha[criterion] = max_values[criterion]
            beta[criterion] = min_values[criterion]

    # each criterion for min value is 0
    for criterion in alternatives.columns:
        problem += vars[criterion + "_" + str(alpha[criterion])] == 0
    
    # sum of max value criteria is 1
    
    problem += pulp.lpSum([vars[criterion + "_" + str(beta[criterion])] for criterion in alternatives.columns]) == 1

    # monotonicity constraints
    for criterion in alternatives.columns:
        vals = alternatives[criterion].unique()
        # sort values in ascending order if gain, descending if not gain
        if gains[criterion]:
            vals = sorted(vals)
        else:
            vals =sorted(vals,reverse=True)

                
        for i in range(len(vals) - 1):
            problem += vars[criterion + "_" + str(vals[i])] <= vars[criterion + "_" + str(vals[i + 1])]

    return problem

import matplotlib.pyplot as plt
import numpy as np

def plot_marginal_values(vars):
    # get unique values of each criterion
    unique_values = dict()
    for criterion in alternatives.columns:
        unique_values[criterion] = alternatives[criterion].unique()
        unique_values[criterion].sort()

    # get marginal values of each criterion
    marginal_values = dict()
    for criterion in alternatives.columns:
        marginal_values[criterion] = [vars[criterion + "_" + str(value)].varValue for value in unique_values[criterion]]

    # plot marginal values
    # subplots for each criterion
    fig, axs = plt.subplots(len(alternatives.columns), 1, figsize=(10, 10))
    for i, criterion in enumerate(alternatives.columns):
        axs[i].plot(unique_values[criterion], marginal_values[criterion])
        # increase margin between plots
        plt.subplots_adjust(hspace=0.5)
        axs[i].set_xlabel(criterion)
        axs[i].set_ylabel("Marginal Value")
        axs[i].grid()
    plt.show()

# for each pairwise comparison add binary constraint


def find_inconsistencies(alternatives,comparisons,gains):
    
    criteria_vars = criterion_values_variables(alternatives)
    problem = pulp.LpProblem("inconsistency",sense = pulp.LpMinimize)
    
    problem = criterion_values_constraints(alternatives,criteria_vars,problem,gains)
    vars = []
    for left, right in comparisons:
        var = pulp.LpVariable(f"var_{left}_{right}",cat=pulp.const.LpBinary)
        vars.append(var)
        #left_side
        left_side = [criteria_vars[criterion + "_" + str(alternatives[criterion][left])] for criterion in alternatives.columns]
        right_side = [criteria_vars[criterion + "_" + str(alternatives[criterion][right])] for criterion in alternatives.columns]
        problem += left_side >= right_side - var + 0.0001
    problem += pulp.lpSum(vars)

    problem.solve()
    def resolve_inconsistencies(problem,vars):

        inconsistencies = []
        for var in vars:
            if var.varValue == 1:
                #print(var.name, var.varValue)
                inconsistencies.append(var)
        problem += pulp.lpSum(inconsistencies) <= len(inconsistencies) -1
        problem.solve()
        return problem,inconsistencies
        
    subsets = []
    problem, inconsistencies = resolve_inconsistencies(problem,vars)
    subsets.append(inconsistencies)
    
    while problem.status > 0:
            problem, inconsistencies = resolve_inconsistencies(problem,vars)
            subsets.append(inconsistencies)

   # print(subsets)
    return subsets
            

subsets = find_inconsistencies(alternatives,comparisons,gains)

def pairwise_matrix(rankings):
    n = len(rankings)
    matrix = [[0]*n for i in range(n)]
    for i in range(n):
        for j in range(i+1, n):
            if rankings.index(i) < rankings.index(j):
                matrix[i][j] = 1
            else:
                matrix[j][i] = 1
    return matrix

def kendalls_tau(matrix1, matrix2):
    kendalls_dist = 0
    for row1, row2 in zip(matrix1, matrix2):
        for i in range(len(row1)):
            kendalls_dist += abs(row1[i] - row2[i])
    kendalls_dist *= 0.5
    tau = 1-4*kendalls_dist/(len(matrix1)*(len(matrix1)-1))
    return tau
